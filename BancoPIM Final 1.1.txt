GO
USE [dbBancoPIM]
GO

--Tabelas
CREATE TABLE Crianca(
idMatricula NUMERIC (4) IDENTITY PRIMARY KEY,
nome VARCHAR (50),
sexo VARCHAR (1),
dtNascimento DATE,
dtMatricula DATE,
observacoes VARCHAR (100),
idTurma NUMERIC (4),
idSala NUMERIC (4),
cpfResponsavel VARCHAR (14) 
);

CREATE TABLE HistoricoCrianca(
idHistorico NUMERIC (4) IDENTITY PRIMARY KEY,
dataAlteracao DATETIME,
alteracao VARCHAR (60),
idMatricula NUMERIC (4)
);

CREATE TABLE Responsavel(
cpf varchar (14) PRIMARY KEY,
nome VARCHAR (50),
dataNascimento DATE,
sexo VARCHAR(1),
endereco VARCHAR (100),
numero NUMERIC (5),
bairro VARCHAR (60),
telefone1 VARCHAR (15),
telefone2 VARCHAR (15),
email VARCHAR (40)
);

CREATE TABLE Cuidador(
cpf varchar (14) PRIMARY KEY,
nome VARCHAR(50),
endereco VARCHAR (60),
numero NUMERIC (5),
bairro VARCHAR (60),
sexo VARCHAR (9),
telefone1 VARCHAR (15),
telefone2 VARCHAR (15),
email varchar (40),
dtNascimento DATE,
dtAdmissao DATE,
idTurma NUMERIC (4),
idSala NUMERIC (4)
);

CREATE TABLE Turma(
idTurma NUMERIC (4) IDENTITY PRIMARY KEY,
periodo VARCHAR (10),
);

CREATE TABLE Sala(
idSala NUMERIC (4) IDENTITY PRIMARY KEY,
nome VARCHAR (11),
);
